$(function () { $('#testForm').on('change', 'input[type="text"]', function (event) { var $nextTr = $(event.target).closest('tr').nextAll('tr:not(.group)').first(); $nextTr.find('input').first().focus(); }); $('#testForm').on('blur', 'input[type="text"]', function (event) { var $nextTd = $(event.target).closest('td').nextAll('td:not(.group)').first(); $nextTd.find('input').first().focus(); }); });
