$(function () { var $form = $('#inputform'); $form.on('change', 'input[type="text"]', function (event) { var $nextTr = $(event.target).closest('tr').nextAll('tr:not(.group)').first(); $nextTr.find('input').first().focus(); }); $form.on('blur', 'input[type="text"]', function (event) { var $nextTd = $(event.target).closest('td').nextAll('td:not(.group)').first(); $nextTd.find('input').first().focus(); }); $form.on('keydown', 'input', function (event) { if (event.which === 13) { event.preventDefault(); var $this = $(event.target); var index = parseFloat($this.attr('tabindex')); if (!isNaN(index)) { $('[tabindex="' + (index + 1) + '"]').focus(); } } }); });
